fn delay_ms
	mem
		$len I 0
		$now I 0
		$start I 0
		$mid I 0
	end_mem
	timer $start
	lb loop
		timer $now
		sub I $now $start
		push I $mid
		ge I $mid $len
		jmp_t finish
	jmp loop
	lb finish
		ret
endfn
		
main
	mem
		$pio I 0
	end_mem
	mov I $pio 6
	lb loop_main
		gpio_write $pio 1
		pushp I 100 
		call delay_ms
		gpio_write $pio 0
		pushp I 100 
		call delay_ms
	jmp loop_main
	hlt
endmain
