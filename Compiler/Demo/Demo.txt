fn final
	mem
		$x I 0
		$y I 0
	end_mem
	mul I $x $y
	ret
endfn

fn test
	mem
		$a I 0
		$b I 0
	end_mem
	add I $a $b
	ret
endfn

fn test2
	mem
		$a I 0
		$b I 0
		$c I 0
	end_mem
	sub I $a $b
	push I $a
	sub I $a $c
	ret
endfn

main
	mem
		$x I 0
		$y I 0
		$z I 0
	end_mem
	mov I $x 10
	mov I $y 11
	le I $x $y
	jmp_t bb
		pushp I 3
		pushp I 2
		pushp I 1
		call test2
		jmp cc
	lb bb
		pushp I $x
		pushp I $y
		call test
	lb cc
	pushp I 5
	pushp I 30
	call final
	hlt
endmain
